CREATE DATABASE mi_base_de_datos;
USE mi_base_de_datos;
CREATE TABLE Categorias ( id_categoria INT PRIMARY KEY AUTO_INCREMENT, nombre_categoria VARCHAR(100) NOT NULL );
CREATE TABLE Productos ( id_producto INT PRIMARY KEY AUTO_INCREMENT, nombre_producto VARCHAR(150) NOT NULL, precio DECIMAL(10, 2), id_categoria INT, FOREIGN KEY (id_categoria) REFERENCES Categorias(id_categoria) );
