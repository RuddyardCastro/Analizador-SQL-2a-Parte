CREATE DATABASE MiBaseDatos;

USE MiBaseDatos;

CREATE TABLE Empleados (
    id INT PRIMARY KEY,
    nombres VARCHAR(50),
    apellidos VARCHAR(50),
    salario INT
);

SELECT * FROM empleados;

SELECT nombres
FROM empleados;

SELECT nombres, apellidos
FROM empleados;

SELECT
  nombres,
  apellidos,
  salario
FROM empleados
WHERE salario > 3800;

SELECT
  nombres,
  apellidos
FROM empleados
WHERE nombre = 'Lucía';

SELECT
 nombres,
 apellidos
FROM empleados
ORDER BY apellidos;

SELECT
  nombres,
  apellidos
FROM empleados
ORDER BY apellidos DESC;

SELECT
  nombres,
  apellidos,
  salario
FROM empleados
ORDER BY salario DESC, apellidos ASC;
